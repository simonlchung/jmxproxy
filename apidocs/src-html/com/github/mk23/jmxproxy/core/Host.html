001package com.github.mk23.jmxproxy.core;
002
003import com.fasterxml.jackson.core.JsonGenerator;
004import com.fasterxml.jackson.core.JsonProcessingException;
005import com.fasterxml.jackson.databind.JsonSerializable;
006import com.fasterxml.jackson.databind.SerializerProvider;
007import com.fasterxml.jackson.databind.jsontype.TypeSerializer;
008
009import java.io.IOException;
010
011import java.util.HashMap;
012import java.util.Map;
013import java.util.Set;
014
015/**
016 * <p>JMX Host tracker and serializer.</p>
017 *
018 * Maintains a map of JMX {@link MBean} names and their values. Implements
019 * JsonSerializable interface to convert the stored map into JSON.
020 *
021 * @see <a href="https://fasterxml.github.io/jackson-databind/javadoc/2.6/com/fasterxml/jackson/databind/JsonSerializable.html">com.fasterxml.jackson.databind.JsonSerializable</a>
022 *
023 * @since   2015-05-11
024 * @author  mk23
025 * @version 3.2.0
026 */
027public class Host implements JsonSerializable {
028    private final Map<String, MBean> mbeans;
029    private final ThreadLocal<Integer> limit = new ThreadLocal<Integer>() {
030        @Override protected Integer initialValue() {
031            return -1;
032        }
033    };
034
035    /**
036     * <p>Default constructor.</p>
037     *
038     * Creates a map of {@link MBean} name to associated values.
039     */
040    public Host() {
041        mbeans = new HashMap<String, MBean>();
042    }
043
044    /**
045     * <p>Inserts a new MBean name to value association.</p>
046     *
047     * Creates a new {@link MBean} object and inserts into the map store
048     * associating it to the specified mbean name.
049     *
050     * @param mbeanName name of the {@link MBean} used as the map key.
051     *
052     * @return the newly created {@link MBean} object.
053     */
054    public final MBean addMBean(final String mbeanName) {
055        if (!mbeans.containsKey(mbeanName)) {
056            MBean mbean = new MBean();
057            mbeans.put(mbeanName, mbean);
058        }
059
060        return mbeans.get(mbeanName);
061    }
062
063    /**
064     * <p>Deletes an mbean association.</p>
065     *
066     * Removes an associated {@link MBean} from the map store for this host.
067     *
068     * @param mbeanName name of the {@link MBean} to remove.
069     */
070    public final void removeMBean(final String mbeanName) {
071        mbeans.remove(mbeanName);
072    }
073
074    /**
075     * <p>Sets the thread local history request limit.</p>
076     *
077     * Set the thread local limit for all {@link com.github.mk23.jmxproxy.util.History} when
078     * serializing to JSON. Because this method returns its object, requesting serialization
079     * can be done * with a single statement.
080     *
081     * <p>For example:</p>
082     *
083     * <code>return Response.ok(host.setLimit(5)).build();</code>
084     *
085     * @see <a href="http://docs.oracle.com/javaee/7/api/javax/ws/rs/core/Response.html">javax.ws.rs.core.Response</a>
086     *
087     * @param bound the number of items to retreive from {@link com.github.mk23.jmxproxy.util.History}
088     *     for this thread.
089     *
090     * @return this host object for chaining calls.
091     */
092    public final Host setLimit(final Integer bound) {
093        limit.set(bound);
094        return this;
095    }
096
097    /**
098     * <p>Getter for mbean names.</p>
099     *
100     * Extracts and returns the unique {@link Set} of all currently stored {@link MBean} names.
101     *
102     * @return {@link Set} of {@link MBean} name {@link String}s.
103     */
104    public final Set<String> getMBeans() {
105        return mbeans.keySet();
106    }
107
108    /**
109     * <p>Getter for specific mbean.</p>
110     *
111     * Fetches the specified {@link MBean} from the map store.
112     *
113     * @param mbean name of the {@link MBean} to look up in the map store.
114     *
115     * @return {@link MBean} object if found, null otherwise.
116     */
117    public final MBean getMBean(final String mbean) {
118        return mbeans.get(mbean);
119    }
120
121    /** {@inheritDoc} */
122    public final void serialize(
123        final JsonGenerator jgen,
124        final SerializerProvider sp
125    ) throws IOException, JsonProcessingException {
126        buildJson(jgen);
127    }
128
129    /** {@inheritDoc} */
130    public final void serializeWithType(
131        final JsonGenerator jgen,
132        final SerializerProvider sp,
133        final TypeSerializer ts
134    ) throws IOException, JsonProcessingException {
135        buildJson(jgen);
136    }
137
138    private void buildJson(final JsonGenerator jgen) throws IOException, JsonProcessingException {
139        int bound = limit.get();
140
141        jgen.writeStartObject();
142        for (Map.Entry<String, MBean> mbeanEntry : mbeans.entrySet()) {
143            jgen.writeObjectField(mbeanEntry.getKey(), mbeanEntry.getValue().setLimit(bound));
144        }
145        jgen.writeEndObject();
146    }
147}
