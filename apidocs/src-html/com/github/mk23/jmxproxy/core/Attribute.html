001package com.github.mk23.jmxproxy.core;
002
003import com.fasterxml.jackson.core.JsonGenerator;
004import com.fasterxml.jackson.core.JsonParser;
005import com.fasterxml.jackson.core.JsonParseException;
006import com.fasterxml.jackson.core.JsonProcessingException;
007import com.fasterxml.jackson.core.JsonToken;
008import com.fasterxml.jackson.databind.JsonNode;
009import com.fasterxml.jackson.databind.JsonSerializable;
010import com.fasterxml.jackson.databind.ObjectMapper;
011import com.fasterxml.jackson.databind.SerializerProvider;
012import com.fasterxml.jackson.databind.jsontype.TypeSerializer;
013
014import java.io.IOException;
015
016import java.lang.reflect.Array;
017
018import java.util.List;
019import java.util.ArrayList;
020
021import javax.management.openmbean.CompositeData;
022import javax.management.openmbean.TabularData;
023
024/**
025 * <p>JMX Attribute tracker and serializer.</p>
026 *
027 * Saves a JMX Attribute value object. Implements JsonSerializable
028 * interface to convert the stored value into JSON. On serialization,
029 * recursively inspects the value type and marshals it to JSON using
030 * the supplied JsonGenerator. For {@link Array}, {@link Iterable},
031 * {@link TabularData}, builds a JSON array.  For {@link CompositeData},
032 * builds a JSON object. For any other native types or null values, builds
033 * a JSON equivalent.
034 *
035 * <p>Special cases:</p>
036 *
037 * <ul>
038 *   <li>
039 *   Any {@link String} that contains valid JSON, will also be recursively
040 *   serialized using a dynamic JsonParser.
041 *   </li>
042 *   <li>
043 *   Any <code>NaN</code> {@link Double} or {@link Float} value will yield
044 *   a JSON string <code>"NaN"</code>.
045 *   </li>
046 *   <li>
047 *   Any <code>infinite</code> {@link Double} or {@link Float} value will
048 *   yield a JSON string <code>"Infinity"</code>.
049 *   </li>
050 * </ul>
051 *
052 * @see <a href="http://fasterxml.github.io/jackson-core/javadoc/2.6/com/fasterxml/jackson/core/JsonGenerator.html">com.fasterxml.jackson.core.JsonGenerator</a>
053 * @see <a href="http://fasterxml.github.io/jackson-core/javadoc/2.6/com/fasterxml/jackson/core/JsonParser.html">com.fasterxml.jackson.core.JsonParser</a>
054 * @see <a href="https://fasterxml.github.io/jackson-databind/javadoc/2.6/com/fasterxml/jackson/databind/JsonSerializable.html">com.fasterxml.jackson.databind.JsonSerializable</a>
055 *
056 * @since   2015-05-11
057 * @author  mk23
058 * @version 3.2.0
059 */
060public class Attribute implements JsonSerializable {
061    private Object attributeValue;
062
063    /**
064     * <p>Default constructor.</p>
065     *
066     * Saves the JMX Attribute object for later serialization.
067     *
068     * @param attributeValue object for later serialization.
069     */
070    public Attribute(final Object attributeValue) {
071        this.attributeValue = attributeValue;
072    }
073
074    /** <p>Getter for attributeValue.</p>
075     *
076     * @return JMX Attribute value object.
077     */
078    public final Object getAttributeValue() {
079        return attributeValue;
080    }
081
082    /** {@inheritDoc} */
083    @Override
084    public final void serialize(
085        final JsonGenerator jgen,
086        final SerializerProvider sp
087    ) throws IOException, JsonProcessingException {
088        buildJson(jgen, attributeValue);
089    }
090
091    /** {@inheritDoc} */
092    @Override
093    public final void serializeWithType(
094        final JsonGenerator jgen,
095        final SerializerProvider sp,
096        final TypeSerializer ts
097    ) throws IOException, JsonProcessingException {
098        buildJson(jgen, attributeValue);
099    }
100
101    /**
102     * <p>JMX Attribute value JSON serializer.</p>
103     *
104     * Inspects the stored JMX Attribute value type and serializes it to
105     * JSON using the supplied JsonGenerator. Recursively calls itself
106     * if finding JMX collections.  For an {@link Array}, {@link Iterable},
107     * or {@link TabularData}, builds a JSON array.  For {@link CompositeData},
108     * builds a JSON object.  For any other native types or null values, builds
109     * a JSON equivalent.
110     *
111     * <p>Special cases:</p>
112     *
113     * <ul>
114     *   <li>
115     *   Any {@link String} that contains valid JSON, will also be recursively
116     *   serialized using a dynamic JsonParser.
117     *   </li>
118     *   <li>
119     *   Any <code>NaN</code> {@link Double} or {@link Float} value will yield
120     *   a JSON string <code>"NaN"</code>.
121     *   </li>
122     *   <li>
123     *   Any <code>infinite</code> {@link Double} or {@link Float} value will
124     *   yield a JSON string <code>"Infinity"</code>.
125     *   </li>
126     * </ul>
127     *
128     * @see <a href="http://fasterxml.github.io/jackson-core/javadoc/2.6/com/fasterxml/jackson/core/JsonGenerator.html">com.fasterxml.jackson.core.JsonGenerator</a>
129     * @see <a href="http://fasterxml.github.io/jackson-core/javadoc/2.6/com/fasterxml/jackson/core/JsonParser.html">com.fasterxml.jackson.core.JsonParser</a>
130     *
131     * @param jgen The jersey-supplied JSON generator to use for serialization.
132     * @param objectValue The JMX Attribute value or an element of a collection to serialize.
133     */
134    private void buildJson(
135        final JsonGenerator jgen,
136        final Object objectValue
137    ) throws IOException, JsonProcessingException {
138        if (objectValue == null) {
139            jgen.writeNull();
140        } else if (objectValue instanceof Boolean) {
141            jgen.writeBoolean((Boolean) objectValue);
142        } else if (objectValue instanceof JsonNode) {
143            jgen.writeTree((JsonNode) objectValue);
144        } else if (objectValue.getClass().isArray()) {
145            jgen.writeStartArray();
146            int length = Array.getLength(objectValue);
147            for (int i = 0; i < length; i++) {
148                buildJson(jgen, Array.get(objectValue, i));
149            }
150            jgen.writeEndArray();
151        } else if (objectValue instanceof Iterable) {
152            Iterable data = (Iterable) objectValue;
153            jgen.writeStartArray();
154            for (Object objectEntry : data) {
155                buildJson(jgen, objectEntry);
156            }
157            jgen.writeEndArray();
158        } else if (objectValue instanceof TabularData) {
159            TabularData data = (TabularData) objectValue;
160            jgen.writeStartArray();
161            for (Object objectEntry : data.values()) {
162                buildJson(jgen, objectEntry);
163            }
164            jgen.writeEndArray();
165        } else if (objectValue instanceof CompositeData) {
166            CompositeData data = (CompositeData) objectValue;
167            jgen.writeStartObject();
168            for (String objectEntry : data.getCompositeType().keySet()) {
169                jgen.writeFieldName(objectEntry);
170                buildJson(jgen, data.get(objectEntry));
171            }
172            jgen.writeEndObject();
173        } else if (objectValue instanceof Number) {
174            Double data = ((Number) objectValue).doubleValue();
175            if (data.isNaN()) {
176                jgen.writeString("NaN");
177            } else if (data.isInfinite()) {
178                jgen.writeString("Infinity");
179            } else {
180                jgen.writeNumber(((Number) objectValue).toString());
181            }
182        } else {
183            try {
184                String input = objectValue.toString();
185                List<JsonNode> parts = new ArrayList<JsonNode>();
186                ObjectMapper mapper = new ObjectMapper();
187                JsonParser parser = mapper.getFactory().createParser(input);
188
189                for (parser.nextToken(); parser.hasCurrentToken(); parser.nextToken()) {
190                    JsonToken tok = parser.getCurrentToken();
191                    long pos = parser.getTokenLocation().getCharOffset();
192                    if (tok == JsonToken.START_OBJECT || tok == JsonToken.START_ARRAY) {
193                        parser.skipChildren();
194                    }
195                    long end = parser.getTokenLocation().getCharOffset()
196                             + parser.getTextLength()
197                             + (tok == JsonToken.VALUE_STRING ? 2 : 0);
198                    parts.add(mapper.readTree(input.substring((int) pos, (int) end)));
199                }
200
201                if (parts.isEmpty()) {
202                    jgen.writeString("");
203                } else if (parts.size() == 1) {
204                    buildJson(jgen, parts.get(0));
205                } else {
206                    buildJson(jgen, parts);
207                }
208            } catch (JsonParseException e) {
209                jgen.writeString(objectValue.toString());
210            }
211        }
212    }
213}
