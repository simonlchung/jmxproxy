001package com.github.mk23.jmxproxy.core;
002
003import com.github.mk23.jmxproxy.util.History;
004
005import com.fasterxml.jackson.core.JsonGenerator;
006import com.fasterxml.jackson.core.JsonProcessingException;
007import com.fasterxml.jackson.databind.JsonSerializable;
008import com.fasterxml.jackson.databind.SerializerProvider;
009import com.fasterxml.jackson.databind.jsontype.TypeSerializer;
010
011import java.io.IOException;
012
013import java.util.Collections;
014import java.util.HashMap;
015import java.util.List;
016import java.util.Map;
017import java.util.Set;
018
019/**
020 * <p>JMX MBean tracker and serializer.</p>
021 *
022 * Maintains a map of JMX {@link Attribute} names and a {@link History} of
023 * their values.  Implements JsonSerializable interface to convert the
024 * stored map into JSON.
025 *
026 * @see <a href="https://fasterxml.github.io/jackson-databind/javadoc/2.6/com/fasterxml/jackson/databind/JsonSerializable.html">com.fasterxml.jackson.databind.JsonSerializable</a>
027 *
028 * @since   2015-05-11
029 * @author  mk23
030 * @version 3.2.0
031 */
032public class MBean implements JsonSerializable {
033    private final Map<String, History<Attribute>> attributes;
034    private final ThreadLocal<Integer> limit = new ThreadLocal<Integer>() {
035        @Override
036        protected Integer initialValue() {
037            return -1;
038        }
039    };
040
041    /**
042     * <p>Default constructor.</p>
043     *
044     * Creates a map of {@link Attribute} name to {@link History} of associated values.
045     */
046    public MBean() {
047        attributes = new HashMap<String, History<Attribute>>();
048    }
049
050    /**
051     * <p>Inserts a new Attribute name to History association.</p>
052     *
053     * Optionally creates a new {@link History} object and inserts into the map store associating
054     * it to the specified attribute name. Appends the specified {@link Attribute} value to the
055     * {@link History}.
056     *
057     * @param attributeName name of the {@link Attribute} used as the map key.
058     * @param attributeObject value of the {@link Attribute} added to history.
059     * @param historySize number of items to preserve in the associated {@link History}.
060     */
061    public final void addAttribute(
062        final String attributeName,
063        final Object attributeObject,
064        final int historySize
065    ) {
066        if (!attributes.containsKey(attributeName)) {
067            attributes.put(attributeName, new History<Attribute>(historySize));
068        }
069
070        attributes.get(attributeName).add(new Attribute(attributeObject));
071    }
072
073    /**
074     * <p>Sets the thread local history request limit.</p>
075     *
076     * Set the thread local limit for all {@link History} when serializing to JSON.
077     * Because this method returns its object, requesting serialization can be done
078     * with a single statement.
079     *
080     * <p>For example</p>
081     *
082     * <code>return Response.ok(mbean.setLimit(5)).build();</code>
083     *
084     * @see <a href="http://docs.oracle.com/javaee/7/api/javax/ws/rs/core/Response.html">javax.ws.rs.core.Response</a>
085     *
086     * @param bound the number of items to retreive from {@link History} for this thread.
087     *
088     * @return this mbean object for chaining calls.
089     */
090    public final MBean setLimit(final Integer bound) {
091        limit.set(bound);
092        return this;
093    }
094
095    /**
096     * <p>Tester for attribute name.</p>
097     *
098     * Checks for requested {@link Attribute} name in the map store.
099     *
100     * @param attribute name of the {@link Attribute} to look up in the map store.
101     *
102     * @return true if the {@link Attribute} exists in the map store, false otherwise.
103     */
104    public final boolean hasAttribute(final String attribute) {
105        return attributes.containsKey(attribute);
106    }
107
108    /**
109     * <p>Getter for attribute names.</p>
110     *
111     * Extracts and returns the unique {@link Set} of all currently stored {@link Attribute} names.
112     *
113     * @return {@link Set} of {@link Attribute} name {@link String}s.
114     */
115    public final Set<String> getAttributes() {
116        return attributes.keySet();
117    }
118
119    /**
120     * <p>Getter for most recent attribute.</p>
121     *
122     * Fetches the most recent {@link Attribute} value for the specified name from the
123     * associated {@link History} in the map store.
124     *
125     * @param attribute name of the {@link Attribute} to look up in the map store.
126     *
127     * @return latest {@link Attribute} object from {@link History} if found, null otherwise.
128     */
129    public final Attribute getAttribute(final String attribute) {
130        if (hasAttribute(attribute)) {
131            return attributes.get(attribute).getLast();
132        }
133
134        return null;
135    }
136
137    /**
138     * <p>Getter for a subset of historical attribute values.</p>
139     *
140     * Fetches an array, limited by the requested bound, of most recent {@link Attribute} values
141     * for the specified name from the associated {@link History} in the map store.
142     *
143     * @param attribute name of the {@link Attribute} to look up in the map store.
144     * @param bound size of the resulting array or full history if this exceeds capacity or less than 1.
145     *
146     * @return {@link List} of the latest {@link Attribute} objects from {@link History} if found, empty otherwise.
147     */
148    public final List<Attribute> getAttributes(final String attribute, final int bound) {
149        if (hasAttribute(attribute)) {
150            return attributes.get(attribute).get(bound);
151        }
152
153        return Collections.emptyList();
154    }
155
156    /** {@inheritDoc} */
157    @Override
158    public final void serialize(
159        final JsonGenerator jgen,
160        final SerializerProvider sp
161    ) throws IOException, JsonProcessingException {
162        buildJson(jgen);
163    }
164
165    /** {@inheritDoc} */
166    @Override
167    public final void serializeWithType(
168        final JsonGenerator jgen,
169        final SerializerProvider sp,
170        final TypeSerializer ts
171    ) throws IOException, JsonProcessingException {
172        buildJson(jgen);
173    }
174
175    private void buildJson(final JsonGenerator jgen) throws IOException, JsonProcessingException {
176        int bound = limit.get();
177
178        jgen.writeStartObject();
179        for (Map.Entry<String, History<Attribute>> attributeEntry : attributes.entrySet()) {
180            if (bound < 0) {
181                jgen.writeObjectField(attributeEntry.getKey(), attributeEntry.getValue().getLast());
182            } else {
183                jgen.writeObjectField(attributeEntry.getKey(), attributeEntry.getValue().get(bound));
184            }
185        }
186        jgen.writeEndObject();
187    }
188}
