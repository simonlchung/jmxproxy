001package com.github.mk23.jmxproxy.jmx;
002
003import com.github.mk23.jmxproxy.core.Host;
004import com.github.mk23.jmxproxy.core.MBean;
005
006import java.io.IOException;
007
008import java.net.MalformedURLException;
009
010import java.util.HashMap;
011import java.util.HashSet;
012import java.util.Map;
013import java.util.Set;
014
015import java.util.concurrent.CountDownLatch;
016import java.util.concurrent.Executors;
017import java.util.concurrent.ScheduledExecutorService;
018import java.util.concurrent.TimeUnit;
019
020import javax.management.MBeanAttributeInfo;
021import javax.management.MBeanServerConnection;
022import javax.management.ObjectName;
023
024import javax.management.remote.JMXConnector;
025import javax.management.remote.JMXConnectorFactory;
026import javax.management.remote.JMXServiceURL;
027
028import org.slf4j.Logger;
029import org.slf4j.LoggerFactory;
030
031/**
032 * <p>JMX fetcher and tracker.</p>
033 *
034 * Creates a {@link ScheduledExecutorService} to periodically make to a {@link JMXConnector}
035 * to a JMX endpoint.  At every execution, discovers and fetches all remote registered mbeans
036 * and populates the local {@link Host} object with assicated {@link MBean}s and their
037 * {@link com.github.mk23.jmxproxy.core.Attribute}.  Also maintains the {@link Host} access
038 * time to allow reaping of
039 * unaccessed workers.
040 *
041 * @since   2015-05-11
042 * @author  mk23
043 * @version 3.2.0
044 */
045public class ConnectionWorker {
046    private static final Logger LOG = LoggerFactory.getLogger(ConnectionWorker.class);
047
048    private Host host;
049    private ConnectionCredentials authCreds;
050
051    private boolean authValid;
052    private IOException connError;
053
054    private final Object fetchLock;
055    private final int historySize;
056    private final long cacheDuration;
057
058    private final JMXServiceURL url;
059
060    private long accessTime;
061
062    private ScheduledExecutorService pollerSvc;
063
064    /**
065     * <p>Default constructor.</p>
066     *
067     * Initializes the {@link JMXServiceURL} to the specified hostName
068     * and starts the {@link ScheduledExecutorService} for periodically
069     * connecting and fetching JMX objects.
070     *
071     * @param hostName host:port {@link String} JMX agent target.
072     * @param cacheDuration period in milliseconds for how often to connect to the JMX agent.
073     * @param historySize number of {@link com.github.mk23.jmxproxy.core.Attribute}s to keep
074     *     per {@link MBean} {@link com.github.mk23.jmxproxy.util.History}.
075     *
076     * @throws MalformedURLException if the specified host is not a valid host:port {@link String}.
077     */
078    public ConnectionWorker(
079        final String hostName,
080        final long cacheDuration,
081        final int historySize
082    ) throws MalformedURLException {
083        this.historySize = historySize;
084        this.cacheDuration = cacheDuration;
085
086        fetchLock = new Object();
087        url = new JMXServiceURL("service:jmx:rmi:///jndi/rmi://" + hostName + "/jmxrmi");
088    }
089
090    /**
091     * <p>Getter for host.<p>
092     *
093     * Fetches the tracked {@link Host} object and resets the request access time.
094     *
095     * @param testCreds optional {@link ConnectionCredentials} for the provided JMX agent or null if none.
096     *
097     * @return {@link Host} as populated by the most recent fetch operation.
098     *
099     * @throws IOException if the specified host is unreachable.
100     * @throws SecurityException if the specified credentials to the host are incorrect.
101     */
102    public final Host getHost(final ConnectionCredentials testCreds) throws IOException, SecurityException {
103        if (host == null) {
104            final CountDownLatch ready = new CountDownLatch(1);
105
106            shutdown();
107
108            host = new Host();
109            authCreds = (testCreds != null) ? testCreds : new ConnectionCredentials();
110
111            pollerSvc = Executors.newSingleThreadScheduledExecutor();
112            pollerSvc.scheduleAtFixedRate(new Runnable() {
113                @Override
114                public void run() {
115                    try {
116                        fetchJMXValues();
117                    } catch (Exception e) {
118                        e.printStackTrace();
119                    }
120                    ready.countDown();
121                }
122            }, 0, cacheDuration, TimeUnit.MILLISECONDS);
123
124            try {
125                ready.await();
126            } catch (InterruptedException e) {
127                LOG.error("unable to finish first run", e);
128            }
129        }
130
131        if (connError != null) {
132            throw connError;
133        } else if (!authCreds.equals(testCreds) && authValid) {
134            throw new SecurityException();
135        } else if (!authValid) {
136            throw new SecurityException();
137        } else {
138            synchronized (fetchLock) {
139                accessTime = System.currentTimeMillis();
140                return host;
141            }
142        }
143    }
144
145    /**
146     * <p>Checks expiration of this worker object.</p>
147     *
148     * Checks the last access time against the provided duration limit to determine if this worker
149     * is expired and can be purged.
150     *
151     * @param accessDuration time in milliseconds of no access after which this worker is expired.
152     *
153     * @return true if this worker hasn't been accessed recently, false otherwise.
154     */
155    public final boolean isExpired(final long accessDuration) {
156        return System.currentTimeMillis() - accessTime > accessDuration;
157    }
158
159    /**
160     * <p>Stops the scheduled fetcher.</p>
161     *
162     * Signals the {@link ScheduledExecutorService} to shutdown for process termination cleanup.
163     */
164    public final void shutdown() {
165        if (pollerSvc != null && !pollerSvc.isShutdown()) {
166            pollerSvc.shutdown();
167        }
168    }
169
170    private void fetchJMXValues() {
171        JMXConnector connection = null;
172        MBeanServerConnection server = null;
173
174        Map<String, Object> environment = new HashMap<String, Object>();
175        if (authCreds != null) {
176            environment.put(JMXConnector.CREDENTIALS, new String[]{authCreds.getUsername(), authCreds.getPassword()});
177        }
178
179        try {
180            LOG.debug("connecting to mbean server " + url);
181
182            synchronized (fetchLock) {
183                connection = JMXConnectorFactory.connect(url, environment);
184                server = connection.getMBeanServerConnection();
185
186                authValid = true;
187                connError = null;
188
189                Set<String> freshMBeans = new HashSet<String>();
190
191                for (ObjectName mbeanName : server.queryNames(null, null)) {
192                    LOG.debug("discovered mbean " + mbeanName);
193                    freshMBeans.add(mbeanName.toString());
194
195                    MBean mbean = host.addMBean(mbeanName.toString());
196                    try {
197                        for (MBeanAttributeInfo attributeObject : server.getMBeanInfo(mbeanName).getAttributes()) {
198                            if (attributeObject.isReadable()) {
199                                try {
200                                    Object attribute = server.getAttribute(mbeanName, attributeObject.getName());
201
202                                    mbean.addAttribute(attributeObject.getName(), attribute, historySize);
203                                } catch (java.lang.NullPointerException e) {
204                                    LOG.error("failed to add attribute " + attributeObject.toString() + ": " + e);
205                                } catch (java.rmi.UnmarshalException e) {
206                                    LOG.error("failed to add attribute " + attributeObject.toString() + ": " + e);
207                                } catch (javax.management.AttributeNotFoundException e) {
208                                    LOG.error("failed to add attribute " + attributeObject.toString() + ": " + e);
209                                } catch (javax.management.MBeanException e) {
210                                    LOG.error("failed to add attribute " + attributeObject.toString() + ": " + e);
211                                } catch (javax.management.RuntimeMBeanException e) {
212                                    if (!(e.getCause() instanceof UnsupportedOperationException)) {
213                                        LOG.error("failed to add attribute " + attributeObject.toString() + ": " + e);
214                                    }
215                                }
216                            }
217                        }
218                    } catch (javax.management.InstanceNotFoundException e) {
219                        LOG.error("failed to get mbean info for " + mbeanName, e);
220                    } catch (javax.management.IntrospectionException e) {
221                        LOG.error("failed to get mbean info for " + mbeanName, e);
222                    } catch (javax.management.ReflectionException e) {
223                        LOG.error("failed to get mbean info for " + mbeanName, e);
224                    }
225                }
226
227                Set<String> staleMBeans = new HashSet<String>(host.getMBeans());
228                staleMBeans.removeAll(freshMBeans);
229                for (String mbeanName : staleMBeans) {
230                    host.removeMBean(mbeanName);
231                    LOG.debug("removed stale mbean " + mbeanName);
232                }
233            }
234        } catch (IOException e) {
235            host = null;
236            connError = e;
237            LOG.error("communication failure with " + url, e);
238        } catch (SecurityException e) {
239            host = null;
240            authValid = false;
241            LOG.error("invalid credentials for " + url, e);
242        } finally {
243            if (connection != null) {
244                try {
245                    connection.close();
246                    LOG.debug("disconnected from " + url);
247                } catch (IOException e) {
248                    LOG.error("failed to disconnect from " + url, e);
249                } finally {
250                    connection = null;
251                }
252            }
253        }
254    }
255}
