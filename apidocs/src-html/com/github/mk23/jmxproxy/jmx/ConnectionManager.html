001package com.github.mk23.jmxproxy.jmx;
002
003import com.github.mk23.jmxproxy.conf.AppConfig;
004import com.github.mk23.jmxproxy.core.Host;
005
006import io.dropwizard.lifecycle.Managed;
007
008import java.net.MalformedURLException;
009
010import java.util.HashMap;
011import java.util.Map;
012import java.util.Set;
013
014import java.util.concurrent.Executors;
015import java.util.concurrent.ScheduledExecutorService;
016import java.util.concurrent.TimeUnit;
017
018import javax.ws.rs.core.Response;
019import javax.ws.rs.WebApplicationException;
020
021import org.slf4j.Logger;
022import org.slf4j.LoggerFactory;
023
024/**
025 * <p>Main application lifecycle object that manages all JMX Agent Connections.</p>
026 *
027 * Implements the dropwizard lifecycle Managed interface to create the object responsible
028 * for all application data management.  Creates workers for connecting to JMX endpoints and
029 * handles requests for data retreival.  Controls purging unaccessed endpoints as well as
030 * their runtime state.
031 *
032 * @see <a href="http://dropwizard.github.io/dropwizard/0.9.2/dropwizard-lifecycle/apidocs/io/dropwizard/lifecycle/Managed.html">io.dropwizard.lifecycle.Managed</a>
033 *
034 * @since   2015-05-11
035 * @author  mk23
036 * @version 3.2.0
037 */
038public class ConnectionManager implements Managed {
039    private static final Logger LOG = LoggerFactory.getLogger(ConnectionManager.class);
040
041    private final AppConfig config;
042
043    private final Map<String, ConnectionWorker> hosts;
044    private final ScheduledExecutorService purge;
045
046    private boolean started = false;
047
048    /**
049     * <p>Default constructor.</p>
050     *
051     * Called by the application initializer.  Creates the map of host:port {@link String}s to
052     * associated {@link ConnectionWorker} instances.  Creates an instance of the unaccessed
053     * endpoints purge thread.  Saves the provided applicaiton configuration for later retreival
054     * request.
055     *
056     * @param config configuration as specified by the administrator at application invocation.
057     */
058    public ConnectionManager(final AppConfig config) {
059        this.config = config;
060
061        hosts = new HashMap<String, ConnectionWorker>();
062        purge = Executors.newSingleThreadScheduledExecutor();
063    }
064
065    /**
066     * <p>Getter for config.</p>
067     *
068     * Fetches the application run-time configuration object.
069     *
070     * @return application configuration.
071     */
072    public final AppConfig getConfiguration() {
073        return config;
074    }
075
076    /**
077     * <p>Getter for hosts.</p>
078     *
079     * Fetches the {@link Set} of {@link ConnectionWorker} name {@link String}s.
080     *
081     * @return {@link Set} of {@link ConnectionWorker} name {@link String}s.
082     */
083    public final Set<String> getHosts() {
084        synchronized (hosts) {
085            return hosts.keySet();
086        }
087    }
088
089    /**
090     * <p>Deleter for host.</p>
091     *
092     * Attempts to remove the specified host from the {@link ConnectionWorker} map store.
093     *
094     * @param host endpoint host:port {@link String}.
095     *
096     * @return true if the key is found in the map store.
097     *
098     * @throws WebApplicationException if key is not found in the map store.
099     */
100    public final boolean delHost(final String host) throws WebApplicationException {
101        synchronized (hosts) {
102            if (hosts.containsKey(host)) {
103                LOG.debug("purging " + host);
104                hosts.remove(host).shutdown();
105
106                return true;
107            } else {
108                throw new WebApplicationException(Response.Status.NOT_FOUND);
109            }
110        }
111    }
112
113    /**
114     * <p>Anonymous getter for host.</p>
115     *
116     * Fetches the specified {@link Host} with anonymous (null) credentials.
117     * Equivalent to calling: <br>
118     *
119     * <code>return getHost(host, null);</code>
120     *
121     * @param host endpoint host:port {@link String}.
122     *
123     * @return {@link Host} object for the requested endpoint.
124     *
125     * @throws WebApplicationException if unauthorized, forbidden, or invalid endpoint.
126     */
127    public final Host getHost(final String host) throws WebApplicationException {
128        return getHost(host, null);
129    }
130
131    /**
132     * <p>Authenticated getter for host.</p>
133     *
134     * Fetches the specified {@link Host} with provided credentials.  Validates endpoint
135     * access against a configured whitelist.  Validates provided credentials against
136     * previous requests and if different, a new {@link ConnectionWorker} object is
137     * instanciated and associated with the specified endpoint.  Lastly, if specified
138     * endpoint is not yet in the map store, instanciates a new {@link ConnectionWorker}
139     * and saves it for later retreival.
140     *
141     * @param host endpoint host:port {@link String}.
142     * @param auth endpoint {@link ConnectionCredentials} or null for anonymous access.
143     *
144     * @return {@link Host} object for the requested endpoint.
145     *
146     * @throws WebApplicationException if
147     *     <ul>
148     *         <li>forbidden (not whitelisted)</li>
149     *         <li>unauthized (incorrect credentials)</li>
150     *         <li>bad request (malformed host:port)</li>
151     *         <li>not found (any other exception)</li>
152     *     </ul>
153     */
154    public final Host getHost(
155        final String host,
156        final ConnectionCredentials auth
157    ) throws WebApplicationException {
158        if (!config.getAllowedEndpoints().isEmpty() && !config.getAllowedEndpoints().contains(host)) {
159            throw new WebApplicationException(Response.Status.FORBIDDEN);
160        }
161
162        try {
163            synchronized (hosts) {
164                if (!hosts.containsKey(host)) {
165                    LOG.info("creating new worker for " + host);
166                    hosts.put(host, new ConnectionWorker(
167                        host,
168                        config.getCacheDuration().toMilliseconds(),
169                        config.getHistorySize()
170                    ));
171
172                }
173                return hosts.get(host).getHost(auth);
174            }
175        } catch (MalformedURLException e) {
176            throw new WebApplicationException(Response.Status.BAD_REQUEST);
177        } catch (SecurityException e) {
178            throw new WebApplicationException(Response.Status.UNAUTHORIZED);
179        } catch (Exception e) {
180            throw new WebApplicationException(Response.Status.NOT_FOUND);
181        }
182    }
183
184    /**
185     * <p>Getter for started.</p>
186     *
187     * Used by the application health check to verify the manager start() method has been invoked.
188     *
189     * @return true if the manager was started, false otherwise.
190     */
191    public final boolean isStarted() {
192        return started;
193    }
194
195    /**
196     * <p>Handler for application startup.</p>
197     *
198     * Starts the unaccessed endpoint purge thread at application initialization.
199     */
200    public final void start() {
201        LOG.info("starting jmx connection manager");
202
203        LOG.debug("allowedEndpoints: " + config.getAllowedEndpoints().size());
204        for (String ae : config.getAllowedEndpoints()) {
205            LOG.debug("    " + ae);
206        }
207
208        long cleanInterval = config.getCleanInterval().toMilliseconds();
209
210        purge.scheduleAtFixedRate(new Runnable() {
211            @Override
212            public void run() {
213                LOG.debug("begin expiring stale hosts");
214                synchronized (hosts) {
215                    for (Map.Entry<String, ConnectionWorker> hostEntry : hosts.entrySet()) {
216                        if (hostEntry.getValue().isExpired(config.getAccessDuration().toMilliseconds())) {
217                            LOG.debug("purging " + hostEntry.getKey());
218                            hosts.remove(hostEntry.getKey()).shutdown();
219                        }
220                    }
221                }
222                LOG.debug("end expiring stale hosts");
223            }
224        }, cleanInterval, cleanInterval, TimeUnit.MILLISECONDS);
225
226        started = true;
227    }
228
229    /**
230     * <p>Handler for application shutdown.</p>
231     *
232     * Stops the purge thread and all currently tracked {@link ConnectionWorker} instances.
233     */
234    public final void stop() {
235        LOG.info("stopping jmx connection manager");
236        purge.shutdown();
237        synchronized (hosts) {
238            for (Map.Entry<String, ConnectionWorker> hostEntry : hosts.entrySet()) {
239                LOG.debug("purging " + hostEntry.getKey());
240                hosts.remove(hostEntry.getKey()).shutdown();
241            }
242        }
243        hosts.clear();
244        started = false;
245    }
246}
