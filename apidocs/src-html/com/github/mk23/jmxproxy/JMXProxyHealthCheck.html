001package com.github.mk23.jmxproxy;
002
003import com.github.mk23.jmxproxy.jmx.ConnectionManager;
004
005import com.codahale.metrics.health.HealthCheck;
006
007/**
008 * <p>Service health check.</p>
009 *
010 * Checks the state of the {@link ConnectionManager}.
011 *
012 * @since   2015-05-11
013 * @author  mk23
014 * @version 3.2.0
015 */
016public class JMXProxyHealthCheck extends HealthCheck {
017    /**
018     * Lifecycle {@link ConnectionManager} object that determines
019     * the overall service status.
020     */
021    private final ConnectionManager manager;
022
023    /**
024     * <p>Default constructor.</p>
025     *
026     * Uses the {@link ConnectionManager} object to determine
027     * service status when a health check is requested.
028     *
029     * @param manager of the endpoint cache.
030     */
031    public JMXProxyHealthCheck(final ConnectionManager manager) {
032        this.manager = manager;
033    }
034
035    /** {@inheritDoc} */
036    @Override
037    protected final Result check() throws Exception {
038        if (!manager.isStarted()) {
039            return Result.unhealthy("connection manager is not started");
040        }
041        return Result.healthy();
042    }
043}
