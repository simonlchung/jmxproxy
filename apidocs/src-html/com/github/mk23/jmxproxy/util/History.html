001package com.github.mk23.jmxproxy.util;
002
003import java.util.ArrayList;
004import java.util.Collections;
005import java.util.List;
006
007/**
008 * <p>Fixed size historical attribute value store.</p>
009 *
010 * Saves an object into a fixed size round robin store.  Allows clients to retreive latest
011 * value, a subset of latest values, or the full stored history of values.  When number
012 * of stored values exceeds the fixed size of the store, the oldest values are overwritten.
013 *
014 * @param <T> type of objects held in this History.
015 *
016 * @since   2015-05-11
017 * @author  mk23
018 * @version 3.3.2
019 */
020public class History<T> {
021    private List<T> objects;
022    private int length;
023    private int pointer;
024    private boolean wrapped;
025
026    /**
027     * <p>Default constructor.</p>
028     *
029     * Initializes the {@link Object} store {@link List} to the given history size.
030     *
031     * @param length number of {@link Object} values to keep in history.
032     */
033    public History(final int length) {
034        this.length = length;
035
036        pointer = -1;
037        objects = new ArrayList<T>(Collections.nCopies(length, (T) null));
038    }
039
040    /**
041     * <p>Adds a item to the store.</p>
042     *
043     * Adds a new item to the history store.  If the number of items exceeds the size
044     * of the store, the oldest item is overwritten.
045     *
046     * @param item newest item to add to the store.
047     */
048    public final void add(final T item) {
049        wrapped = ++pointer == length;
050        pointer = pointer % length;
051
052        objects.set(pointer, item);
053    }
054
055    /**
056     * <p>Gets the latest item from the history store.</p>
057     *
058     * Gets the latest single item from the history store.
059     *
060     * @return Latest item from the history store or null if empty.
061     */
062    public final T getLast() {
063        return pointer < 0 ? null : objects.get(pointer % length);
064    }
065
066    /**
067     * <p>Gets the full item history from the store.</p>
068     *
069     * Gets the full item history from the store as a {@link List}.
070     *
071     * @return {@link List} of the full history.  May be empty if history has no items.
072     */
073    public final List<T> get() {
074        return get(length);
075    }
076
077    /**
078     * <p>Gets a limited item history from the store.</p>
079     *
080     * Gets a limited history from the store as a {@link List}, based
081     * on requested item count.  If the specified count is less than 0 or greater than
082     * the number of entries, a full history is returned instead.
083     *
084     * @param limit number of items to retreive from the history store.
085     *
086     * @return {@link List} of the requested slice of history.  May be empty if history has no items.
087     */
088    public final List<T> get(final int limit) {
089        if (pointer < 0) {
090            return Collections.emptyList();
091        }
092
093        int head = pointer + length * Boolean.compare(wrapped, false);
094        int size = Math.min(limit > 0 ? limit : Integer.MAX_VALUE, Math.min(head + 1, length));
095        List<T> rval = new ArrayList<T>(size);
096
097        for (int i = 0; i < size; i++) {
098            rval.add(objects.get((head - i) % length));
099        }
100
101        return rval;
102    }
103}
