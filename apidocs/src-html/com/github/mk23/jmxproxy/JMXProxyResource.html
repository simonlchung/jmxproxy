471    private Response getJMXHost(
472        final String hostName,
473        final int port,
474        final boolean full,
475        final int limit,
476        final ConnectionCredentials auth
477    ) {
478        Host host = manager.getHost(hostName + ":" + port, auth);
479        return Response.ok(full ? host.setLimit(limit) : host.getMBeans()).build();
480    }
481
482    private Response getJMXHost(
483        final String hostName,
484        final int port,
485        final String mbeanName,
486        final boolean full,
487        final int limit,
488        final ConnectionCredentials auth
489    ) {
490        MBean mbean = manager.getHost(hostName + ":" + port, auth).getMBean(mbeanName);
491        if (mbean == null) {
492            return Response.status(Response.Status.NOT_FOUND).build();
493        }
494
495        return Response.ok(full ? mbean.setLimit(limit) : mbean.getAttributes()).build();
496    }
497
498    private Response getJMXHost(
499        final String hostName,
500        final int port,
501        final String mbeanName,
502        final String attribute,
503        final int limit,
504        final ConnectionCredentials auth
505    ) {
506        MBean mbean = manager.getHost(hostName + ":" + port, auth).getMBean(mbeanName);
507        if (mbean == null || !mbean.hasAttribute(attribute)) {
508            return Response.status(Response.Status.NOT_FOUND).build();
509        }
510
511        if (limit < 0) {
512            return Response.ok(mbean.getAttribute(attribute)).build();
513        } else {
514            return Response.ok(mbean.getAttributes(attribute, limit)).build();
515        }
516    }
